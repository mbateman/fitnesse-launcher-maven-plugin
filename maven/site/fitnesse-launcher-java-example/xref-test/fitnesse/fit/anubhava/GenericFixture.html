142     public void changeSUT()
143     {
144       // check if it is var=.method expression, var=.method means 
145       // calling method for the object represented by var.
146 
147       String text = new String(methodName);
148 
149       int eql = text.indexOf(EQ);
150       if (eql > 0 && eql < (text.length()-1))
151       {
152         int eqldot = text.indexOf("=.", eql);
153         if (eqldot >= 0)
154         {
155           this.methodName = text.substring(eqldot+2);
156 
157           if (eqldot == eql)
158             this.changedSUT = camel(text.substring(0, eqldot));
159           else
160           {
161             this.varName = camel(text.substring(0, eql));
162             this.changedSUT = camel(text.substring(eql+1, eqldot));
163           }
164         }
165         else
166         {
167           this.varName = camel(text.substring(0, eql));
168           this.methodName = text.substring(eql+1);
169         }
170       }
171     }

238   public static String[] exec(String command) {
239     List<String> output = new ArrayList<String>();
240     try {
241       Runtime runtime = Runtime.getRuntime();
242       Process proc = runtime.exec(command);
243 
244       // put a BufferedReader on the shell output
245       InputStream inputstream = proc.getInputStream();
246       InputStreamReader inputstreamreader = new InputStreamReader(inputstream);
247       BufferedReader bufferedreader = new BufferedReader(inputstreamreader);
248 
249       // read the command output
250       String line;
251       while ((line = bufferedreader.readLine()) != null) {
252         output.add(line);
253       }
254     }
255     catch (IOException e) {
256       System.err.println(e);
257     }
258 
259     return ( (String[] ) output.toArray(new String[0]) );
260   }

297   protected RowEntry readRow(Parse row)
298   {
299     List<String> methodParameters = new ArrayList<String>();
300     List<String> returnVals = new ArrayList<String>();
301 
302     RowEntry re = new RowEntry();
303 
304     boolean endOfMethod = false;
305     int c=0;
306     for (Parse cells = row.parts; cells != null; c++, cells=cells.more)
307     {
308       String text = Parse.unescape(Parse.unformat(cells.body).trim());
309 
310       //debug("Adding cell: " + cells);
311       re.cells.add(cells);
312 
313       if (c==0)
314       {
315         if (text.length() == 0 || text.charAt(0) == COMMENT)
316         {
317           re.ignored = true;
318           break;
319         }
320 
321         re.methodName = text;
322         re.changeSUT();
323       }
324       else
325       {
326         // check it this is blank end-of-method cell
327         if (!endOfMethod && text.trim().equals(""))
328         {
329           endOfMethod = true;
330           re.startReturnVals = c+1;
331           continue;
332         }
333 
334         if (!endOfMethod)
335           methodParameters.add(text);
336         else
337         {
338           // is it a static class.field?
339           Object ret = getStaticField(text);
340           if (ret != null)
341             text = ret.toString();
342 
343           returnVals.add(text);
344         }
345       }
346     }
347 
348     if (re.ignored == false)
349     {
350       DSLAdapter.CommandRow cr = 
351         DSLAdapter.findMatch(re.methodName, methodParameters.toArray(new String[0]));
352       if (cr != null)
353       {
354         re.methodName = cr.getMethodName();
355         re.changeSUT();
356         methodParameters = cr.getParameters();
357         debug("Found matching command: " + re.methodName + " "
358             + methodParameters);
359       }
360       else
361         re.methodName = super.camel(re.methodName);
362 
363       re.methodParameters = methodParameters.toArray(new String[0]);
364       re.returnVals = returnVals.toArray(new String[0]);
365     }
366 
367     re.methodParameters = translateSymbols(re.methodParameters);
368     re.returnVals = translateSymbols(re.returnVals);
369 
370     debug(re);
371     return re;
372   }

374   public Object getStaticField(String fieldName)
375   {
376     Object ret = null;
377 
378     if (fieldName == null || fieldName.length() == 0)
379       return ret;
380 
381     // is it a static class.field?
382     int dot = fieldName.lastIndexOf('.');
383     if (dot > 0)
384     {
385       try
386       {
387         //debug("Checking class name: " + fieldName.substring(0, dot));
388         Class<?> fieldType = Class.forName(fieldName.substring(0, dot));
389         if (!fieldName.substring(dot+1).equals("class"))
390         {
391           Field f = fieldType.getField(fieldName.substring(dot+1));
392           //debug("Got static field: " + f);
393           f.setAccessible(true);
394           ret = f.get(null);
395         }
396         else
397           ret = fieldType;
398       }
399       catch (Exception e)
400       {
401         // do nothing
402       }
403     }
404     return ret;
405   }

421   public static boolean hasEqualsMethod(Class<?> type)
422   {
423     boolean isDefined = false;
424     try
425     {
426       if (type != null && !type.equals(Object.class))
427       {
428         Method m = type.getDeclaredMethod("equals",
429             new Class[] {Object.class});
430         if (m.getReturnType().equals(boolean.class))
431           isDefined = true;
432         else
433           isDefined = false;
434       }
435     } catch (NoSuchMethodException e)
436     {
437       isDefined = false;
438     }
439     return isDefined;
440   }

442   public static boolean callEqualsMethod(Class<?> type, Object o1, Object o2)
443   throws Exception
444   {
445     boolean result = false;
446 
447     TypeAdapter ta = TypeAdapter.adapterFor(type);
448     if (ta != null &&
449         !ta.getClass().equals(fit.TypeAdapter.class))
450     {
451       result = ta.equals(o1, ta.parse(o2.toString()));
452     }
453     else
454     {
455       Method m = type.getMethod("equals", 
456           new Class[] {Object.class});
457       Object ret = m.invoke(o1, new Object[] {o2});
458       if (ret.getClass().equals(Boolean.class))
459         result = ((Boolean) ret).booleanValue();
460     }
461     return result;
462   }

464   private Object parseParameter(String parameter, Class<?> paramType)
465   throws Exception
466   {
467     Object paramObject = null;
468     Object v = fetchVariable(parameter);
469     if (v != null)
470     {
471       // its a variable name
472       paramObject = v;
473     }
474     else
475     {
476       // is it a static class.field?
477       Object ret = getStaticField(parameter);
478       if (ret != null)
479         return ret;
480 
481       TypeAdapter ta = TypeAdapter.adapterFor(paramType);
482       //debug(paramType + " Parameter: [" + parameter + "] " + ta.getClass());
483       if (ta != null &&
484           !ta.getClass().equals(fit.TypeAdapter.class)
485       )
486       {
487         //debug("Inside Parameter: [" + parameter + "] " + ta.getClass());
488         try
489         {
490           if (paramType.isArray() && parameter.startsWith(ARRAY))
491           {
492             ta.init(this, paramType);
493             Class<?> componentType = paramType.getComponentType();
494             TypeAdapter cta = TypeAdapter.on(this, componentType);
495             //debug(cta + " - componentType: " + componentType.getCanonicalName());
496 
497             String[] arrStr = parameter.substring(ARRAY.length()).split(",");
498             paramObject = Array.newInstance(componentType, arrStr.length);
499             for (int i=0; i < arrStr.length; i++) {
500               String token = arrStr[i].trim();
501               Object var = fetchVariable(token);
502               if (var == null) {
503                 var = cta.parse(token);
504                 //debug(var + " - i: " + i + " - " + token);
505               }
506               Array.set(paramObject, i, var);
507             }
508           }
509           else
510             paramObject = ta.parse(parameter);
511         }
512         catch (NumberFormatException ne)
513         {
514           throw new NullPointerException(ne.toString());
515         }
516         catch (NullPointerException npe)
517         {
518           throw npe;
519         }
520       }
521       else
522       {
523         // handle user defined objects as the method arguments                       
524         // user defined objects MUST have a method defined as
525         // public static Object parse(String str) { ... }
526         if (super.hasParseMethod(paramType))
527           paramObject = super.parse(parameter, paramType);
528         else if (parameter.startsWith(ARRAY))
529           paramObject = ta.parse(parameter.substring(ARRAY.length()));
530         else
531           paramObject = parameter;
532       }
533     }
534     return paramObject;
535   }

537   /*
538    * Here we try to lookup a class from a given name and
539    * create an instance from given parameters.
540    * Child class can use this method to set an
541    * object on which reflection will invoke all methods
542    */
543   protected Object getTarget() throws Exception
544   {
545     String[] args = getArgs();
546     if (targetClass == null && args != null && args.length > 0)
547     {
548       String className = super.camel(args[0]);
549       int eql = -1;
550       String varName = null;
551 
552       Object v = fetchVariable(className);
553       if (v != null)
554       {
555         // its a variable name
556         this.target = v;
557         this.targetClass = this.target.getClass();
558         return this.target;
559       }
560       else
561       {
562         // check if we want to store SUT in a variable
563         eql = className.lastIndexOf(EQ);
564         if (eql > 0)
565         {
566           varName = className.substring(0, eql);
567           className = className.substring(eql+1);
568         }
569       }
570 
571       this.targetClass = Class.forName(className);
572       Constructor<?>[] constructors = this.targetClass.getConstructors();
573       int numParams = args.length-1;
574 
575       for (Constructor<?> constructor: constructors)
576       {
577         Class<?>[] parameterTypes = constructor.getParameterTypes();
578         if (numParams != parameterTypes.length)
579           continue;
580 
581         //debug("trying to construct " + constructor);
582         Object[] methodParameters = new Object[numParams];
583         boolean nextConst = false;
584         for (int p=0; p < numParams; p++)
585         {
586           try
587           {
588             methodParameters[p] = 
589               parseParameter(args[p+1], parameterTypes[p]);
590           }
591           catch (NullPointerException npe)
592           {
593             nextConst = true;
594             break;
595           }
596         }
597 
598         // try to invoke the next constructor
599         if (nextConst)
600           continue;
601 
602         try
603         {
604           this.target = constructor.newInstance(methodParameters);
605           debug("*** Constructed " + constructor);
606           debug("--------------------------------------------");
607           break;
608         }
609         catch (Exception e)
610         {
611           if (e instanceof InvocationTargetException ||
612               e instanceof IllegalAccessException)
613             throw e;
614         }
615         finally
616         {
617           if (eql > 0)
618             storeVariable(varName, this.target);
619         }
620       }
621     }
622     return this.target;
623   }

625   /*
626    * overriding doRows() method of Fixture class
627    */
628   @Override
629   public void doRows(Parse rows)
630   {
631     //debug("Rows: " + new Table(rows).toString());
632     RowEntry re = null;
633     try
634     {
635       this.target = this.getTarget();
636 
637       if (this.target != null && this.targetClass == null)
638         this.targetClass = this.target.getClass();
639 
640       //still null?
641       if (this.targetClass == null)
642         this.targetClass = this.getClass();
643       //throw new NullPointerException("Target class is null, cannot continue");
644 
645       for ( int r=0; rows != null; r++, rows=rows.more )
646       {
647         re = readRow(rows);
648         //debug("Rows: " + new Table(rows).toString());
649         int numParams = re.methodParameters.length;
650 
651         if (re.ignored)
652         {
653           super.counts.ignores++;
654           continue;
655         }
656         else if (re.methodName == null || re.methodName.length() == 0)
657         {
658           re.wrong(0, "Invalid Method Name");
659           continue;
660         }
661         else if (re.varName != null && re.changedSUT != null &&
662             re.methodName.equals(THIS))
663         {
664           storeVariable(re.varName, this.target);
665           continue;
666         }
667 
668         Object mTarget = null;
669         Class<?> mTargetClass = null;
670         int dot = 0;
671         // is it a static class.method call?
672         if (re.changedSUT == null)
673           dot = re.methodName.lastIndexOf('.');
674         if (dot > 0)
675         {
676           mTarget = null;
677           String[] arr = re.methodName.split("\\.");
678           mTargetClass = Class.forName(re.methodName.substring(0, dot));
679           re.methodName = arr[arr.length-1];
680         }
681         else if (re.changedSUT != null)
682         {
683           // SUT has itself has changed
684           Object o = Fixture.getSymbol(re.changedSUT);
685           if (o == null)
686           {
687             re.exception(new Exception("Variable not found: " + re.changedSUT));
688             continue;
689           }
690           mTarget = o;
691           mTargetClass = o.getClass();
692         }
693         else
694         {
695           mTarget = this.target;
696           mTargetClass = this.targetClass;
697         }
698 
699         // get all the methods for a given class
700         Method[] methods = mTargetClass.getMethods();
701 
702         boolean matched = false;
703 
704         // find a matching method by name
705         for (Method method: methods)
706         {
707           if ( method.getName().equals(re.methodName) )
708           {
709             Class<?>[] parameterTypes = method.getParameterTypes();
710             //debug(numParams + " Verifying " + method + " " + parameterTypes.length);
711             if (numParams != parameterTypes.length)
712               continue;
713 
714             Object[] methodParameters = new Object[numParams];
715             boolean nextMethod = false;
716             for (int p=0; p < numParams; p++)
717             {
718               try
719               {
720                 if (re.methodParameters[p] == null)
721                   methodParameters[p] = null;
722                 else
723                   methodParameters[p] = 
724                     parseParameter(re.methodParameters[p],
725                         parameterTypes[p]);
726               }
727               catch (NullPointerException npe)
728               {
729                 //debug("Got npe: " + npe);
730                 nextMethod = true;
731                 break;
732               }
733             }
734 
735             // try to get the next method
736             if (nextMethod)
737               continue;
738 
739             try
740             {
741               debug("Invoking " + method);
742               method.setAccessible(true);
743               Object returnVal = method.invoke(mTarget, methodParameters);
744               Class<?> returnType = method.getReturnType();
745               handleResult(re, returnVal, returnType);
746               matched = true;
747               break;
748             }
749             catch (Exception e)
750             {
751               if (e instanceof InvocationTargetException ||
752                   e instanceof IllegalAccessException ||
753                   e instanceof NullPointerException)
754                 throw e;
755               else if (!(e instanceof IllegalArgumentException))
756                 doException(e);
757             }
758           }
759         }
760 
761         debug("--------------------------------------------");
762         if (!matched)
763         {
764           String err = "Method [" + mTargetClass.getCanonicalName() + "." +
765           re.methodName + "()] could not be found matching given parameters";
766           if (mTarget == null)
767             err = "Static " + err;
768           throw new NoSuchMethodException(err);
769         }
770       }
771     }
772     catch (Exception e)
773     {
774       RowEntry r = re;
775       if (r == null)
776         r = readRow(rows);
777       doException(r, e);
778     }
779   }

781   /*
782    * result handler
783    * child class may override this method to have their own result handling
784    */
785   protected void handleResult(RowEntry re, Object actual, Class<?> returnType)
786   throws Exception
787   {
788     String returnTypeName = returnType.getName();
789 
790     debug("Actual Return Type: " + returnTypeName);
791     debug("Actual Return Value: [" + actual + ']');
792 
793     if (returnTypeName.equals("void"))
794       return;
795 
796     if (re.varName != null)
797     {
798       if (!re.varName.equals(THIS))
