41    public static class CommandRow {
42      protected String methodName = null;

47      public CommandRow(String expression, String methodName, String seq)
48      {
49        super();
50        this.regexp = Pattern.compile(expression);
51        this.methodName = methodName;
52        this.paramSequence = seq;

55      @Override
56      public String toString()
57      {
58        StringBuffer sb = new StringBuffer();
59        sb.append("Method Name: " + methodName);
60        List<String> params = getParameters();
61        if (params != null && params.size() > 0)
62          sb.append("\nMethod Parameters: " + getParameters());
63        if (matcher != null)
64          sb.append("\nMatcher: " + matcher.pattern());
65        return sb.toString();
66      }

68      private String matchIt(String s)
69      {
70        return matchIt(s, false);
71      }

73      private String matchIt(String s, boolean isMethod)
74      {
75        //System.err.println("s is: " + s);
76        Matcher m = d.matcher(s);
77  
78        StringBuffer sb = new StringBuffer();
79        while (m.find()) {
80          int start = m.start();
81          if (s.charAt(start) == '\\')
82            continue;
83          else
84          {
85            int argNo = Integer.parseInt(m.group(1));
86            if (argNo <= matcher.groupCount())
87            {
88              String str = matcher.group(argNo);
89              //System.err.println("* group is: " + str);
90              if (str.charAt(0) == '\\')
91                sb.append('\\');
92              else if (!isMethod) {
93                Object v = GenericFixture.fetchVariable( str );
94                // resolve variables only for String type
95                if ( v != null && !v.getClass().isArray() 
96                    && ( v.getClass().equals(String.class) || 
97                      v.getClass().getGenericSuperclass().equals(String.class) )
98                   )
99                  str = v.toString();
100             }
101             m.appendReplacement(sb, str);
102             //System.err.println("* sb is: " + sb);
103           }
104         }
105 
106       }
107       m.appendTail(sb);
108 
109       //strip all escape characters now
110       if (s.charAt(0) != '\\')
111         m = none.matcher(sb.toString());
112       else
113         m = backsp.matcher(sb.toString());
114       String param = m.replaceAll("");
115       //System.err.println("* param is: " + param);
116       return  param ;
117     }

129     public List<String> getParameters() {
130       //split params on , ignoring \,
131       List<String> parameters = new ArrayList<String>();
132       if (paramSequence.length() > 0 && this.matcher != null)
133       {
134         ArrayList<String> items = new ArrayList<String>();
135         int prev=0;
136         // special case for bean shell
137         if ( !methodName.equals("GenericFixture.bsh") )
138         {
139           for(int i=0; i<paramSequence.length(); i++)
140           {
141             if (paramSequence.charAt(i) == paramDelim &&
142                 (i==0 || paramSequence.charAt(i-1) != '\\'))
143             {
144               items.add(paramSequence.substring(prev, i).replace("\\,", ",").trim());
145               prev=i+1;
146             }
147           }
148         }
149         //System.err.println("paramSequence: " + paramSequence + " - " + prev + " - " + paramSequence.substring(prev));
150         items.add( paramSequence.substring(prev).replace("\\,", ",").trim() );
151         //System.err.println("items: " + items);
152         // now replace all $1, $2 etc
153         for(String s: items)
154         {
155           parameters.add( matchIt(s) );
156         }
157       }
158       return parameters;
159     }

167   public void readRow(Parse row) throws InvalidParamSequence
168   {
169     int c=0;
170     String key = null;
171     List<String> values = new ArrayList<String>();
172     for (Parse cells = row.parts; cells != null; c++, cells=cells.more)
173     {
174       String text = Parse.unescape(Parse.unformat(cells.body).trim());
175       if (c==0)
176       {
177         key = text;
178       }
179       values.add(text.trim());
180     }
181 
182     if (c<2)
183       throw new InvalidParamSequence("Invalid DSL syntax: " + values);
184 
185     StringBuffer expression = new StringBuffer(key);
186     for(c=1; c<values.size()-2; c++)
187     {
188       expression.append(cellDelim);
189       String val = values.get(c);
190       if (val.equals("%"))
191         expression.append("(\\p{Print}+)");
192       else
193         expression.append(val);
194     }
195     this.add(key, expression.toString(), values.get(c), values.get(c+1));
196   }

220   public static CommandRow findMatch(String key, String... items) {
221     StringBuffer record = new StringBuffer(items.length);
222     record.append(key);
223     for(String s: items) {
224       record.append(cellDelim);
225       record.append(s);
226     }
227 
228     CommandRow cr = null;
229     List<CommandRow> rows = commandMap.get(camel(key));
230     if (rows != null)
231     {
232       for (CommandRow cRow: rows) {
233         //System.err.println("Comparing [" + record + "] with [" +
234         //cRow.getExpression() + "]");
235         if (cRow.matches(record.toString()))
236         {
237           cr = cRow;
238           //System.err.println("Matched: [" + cr + "]");
239           break;
240         }
241       }
242     }
243     return cr;
244   }

246   public void add(String key, String value, String methodName, String seq)
247   {
248     key = camel(key);
249     List<CommandRow> l = commandMap.get(key);
250     if (l == null)
251       commandMap.put(key, l=new ArrayList<CommandRow>());
252     l.add(new CommandRow(value, methodName, seq));
253   }
